================= HUNTER CLI FLAGS =================

[ CORE ]
-t, --target <domain|ip>        : Target utama (domain/IP)
-i, --input <file>              : File berisi banyak target
--scope-file <file>             : Scope yang diperbolehkan
--exclude <file>                : Target yang di-skip
--threads <num>                 : Jumlah thread
--timeout <sec>                 : Timeout request
--rate <num>                    : Request per detik
--proxy <url>                   : Gunakan proxy
--output <file>                 : Simpan hasil
--format json|txt|yaml|html     : Format output
--silent                        : Output minimal
--verbose                       : Output detail
--debug                         : Debug mode

[ TARGET IDENTITY ]
--whois                         : Ambil data WHOIS
--whois-full                    : WHOIS lengkap
--registrar                     : Info registrar
--org                           : Info organisasi
--contacts                      : Email & kontak

[ DNS ]
--dns                           : Query DNS dasar
--dns-bruteforce                : Brute subdomain DNS
--dns-zone-transfer             : Coba AXFR
--dns-records A,AAAA,MX,TXT     : Record spesifik
--dns-history                   : Riwayat DNS
--passive-dns                   : Data DNS pasif

[ NETWORK ]
--port-scan                     : Scan port
--ports 1-65535                 : Range port
--top-ports <num>               : Port populer
--udp                           : Scan UDP
--service-detect                : Deteksi service
--banner                        : Ambil banner
--os-detect                     : Deteksi OS
--traceroute                    : Jalur network

[ WEB & TLS ]

--headers                       : Ambil header
--tls                           : Info TLS dasar
--tls-deep                      : TLS mendalam
--ciphers                       : Cipher suite
--ssl-cert                      : Info sertifikat
--cert-transparency             : Data CT log

[ TECHNOLOGY ]

--tech-deep                     : Versi & detail
--cms-detect                    : CMS (WordPress, dll)
--framework-detect              : Framework backend
--js-libs                       : Library JS

[ SUBDOMAIN & ASSET ]

--sub-active                    : Scan aktif

--asset-discovery               : Cari asset lain

[ FILE & DIRECTORY ]

--dirscan                       : Scan direktori
--filescan                      : Scan file
--sensitive-files               : Cari file sensitif
--backup-files                  : Cari backup

[ AUTH ]

--mfa-detect                    : Deteksi MFA

[ API ]

Contoh:
hunter -t example.com --full --output result.json
hunter -t example.com --api --js --auth --attack-path
hunter -t example.com --cloud --s3-scan --origin-ip
